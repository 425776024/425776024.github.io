        :rtype: nothing

        if len(self.stack2) == 0:

                self.stack2.append(curr)
        self.stack2.pop()

        :rtype: int

        if len(self.stack2) == 0:

        return self.stack2[-1]

        """
        :rtype: bool
        """

class Logger(object):

        Initialize your data structure here.
        """

    def shouldPrintMessage(self, timestamp, message):
        """

        :type message: str
        :rtype: bool

        while len(self.heap):
            if self.heap[0][0] <= timestamp:
                temp = heapq.heappop(self.heap)
                self.cache.pop(temp[1])
            else:
                break
        if timestamp < self.cache.get(message, 0):
            return False
        self.cache[message] = timestamp + 10
        heapq.heappush(self.heap, (timestamp + 10, message))

    def   init  (self):
        """
        Initialize your data structure here.
        """
        self.size = 10000
        self.nodes = [None] * self.size

    def put(self, key, value):
        """
        value will always be non-negative.
        :type key: int
        :type value: int
        :rtype: void
        """
        index = hash(key) % self.size
        if self.nodes[index] is None:
            self.nodes[index] = ListNode(-1, -1)
        prev = find(self.nodes[index], key)
        if prev.next is None:
            prev.next = ListNode(key, value)
        else:
            prev.next.val = value

    def get(self, key):
        """
        Returns the value to which the specified key is mapped, or -1 if this map contains no mapping for the key
        :type key: int
        :rtype: int
        """
        index = hash(key) % self.size
        if self.nodes[index] is None:
            return -1
        prev = find(self.nodes[index], key)
        if prev.next is None:
            return -1
        else:
            return prev.next.val

    def remove(self, key):
        """
        Removes the mapping of the specified value key if this map contains a mapping for the key

        :rtype: void

count1*(A.length-1)!+count2*(A.length-2)!+......+countn*(0)!
